## 1. Project Setup

### Install Dependencies
```bash
pip install django
pip install pillow  # for image handling
pip install django-crispy-forms
pip install django-widget-tweaks
```

### Create requirements.txt
```
Django==4.2.7
Pillow==10.0.1
django-crispy-forms==2.0
django-widget-tweaks==1.5.0
```

### Initialize Project
```bash
django-admin startproject college_management
cd college_management
python manage.py startapp accounts
python manage.py startapp students
python manage.py startapp faculty
python manage.py startapp courses
python manage.py startapp attendance
```

## 4. Views

### accounts/views.py
```python
from django.shortcuts import render, redirect
from django.contrib.auth import login, authenticate
from django.contrib.auth.decorators import login_required
from django.contrib import messages
from django.contrib.auth.forms import AuthenticationForm
from .forms import UserRegistrationForm

def login_view(request):
    if request.method == 'POST':
        form = AuthenticationForm(request, data=request.POST)
        if form.is_valid():
            username = form.cleaned_data.get('username')
            password = form.cleaned_data.get('password')
            user = authenticate(username=username, password=password)
            if user is not None:
                login(request, user)
                return redirect('dashboard')
    else:
        form = AuthenticationForm()
    return render(request, 'accounts/login.html', {'form': form})

@login_required
def dashboard(request):
    context = {
        'user': request.user,
    }
    
    if request.user.user_type == 'student':
        from students.models import Student
        try:
            student = Student.objects.get(user=request.user)
            context['student'] = student
        except Student.DoesNotExist:
            pass
    elif request.user.user_type == 'faculty':
        from faculty.models import Faculty
        try:
            faculty = Faculty.objects.get(user=request.user)
            context['faculty'] = faculty
        except Faculty.DoesNotExist:
            pass
    
    return render(request, 'dashboard/dashboard.html', context)

def register(request):
    if request.method == 'POST':
        form = UserRegistrationForm(request.POST)
        if form.is_valid():
            user = form.save()
            messages.success(request, 'Registration successful!')
            return redirect('login')
    else:
        form = UserRegistrationForm()
    return render(request, 'accounts/register.html', {'form': form})
```

### students/forms.py
```python
from django import forms
from .models import Student
from accounts.models import User

class StudentForm(forms.ModelForm):
    username = forms.CharField(max_length=150)
    first_name = forms.CharField(max_length=30)
    last_name = forms.CharField(max_length=30)
    email = forms.EmailField()
    password = forms.CharField(widget=forms.PasswordInput)
    
    class Meta:
        model = Student
        fields = ['student_id', 'roll_number', 'department', 'year', 'semester', 
                 'date_of_birth', 'guardian_name', 'guardian_phone', 'admission_date']
        widgets = {
            'date_of_birth': forms.DateInput(attrs={'type': 'date'}),
            'admission_date': forms.DateInput(attrs={'type': 'date'}),
        }
    
    def save(self, commit=True):
        # Create user first
        user = User.objects.create_user(
            username=self.cleaned_data['username'],
            first_name=self.cleaned_data['first_name'],
            last_name=self.cleaned_data['last_name'],
            email=self.cleaned_data['email'],
            password=self.cleaned_data['password'],
            user_type='student'
        )
        
        # Create student
        student = super().save(commit=False)
        student.user = user
        if commit:
            student.save()
        return student
```
